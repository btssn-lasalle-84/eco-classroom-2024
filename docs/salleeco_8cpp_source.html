515 void SalleEco::determinerIndiceConfinement()
516 {
517  int indiceConfinementPrecedent = indiceConfinement;
518 
519  if(mesuresCO2.size() > 1)
520  {
521  int n0 = 0; // nombre de valeurs inférieures ou égales à 1000 ppm (n0)
522  int n1 = 0; // nombre de valeurs comprises entre 1000 et 1700 ppm inclus (n1)
523  int n2 = 0; // nombre de valeurs supérieures à 1700 ppm (n2)
524  double f1 = 0.;
525  double f2 = 0.;
526 
527  n1 =
528  calculProportionBasse(); // nombre de valeurs comprises entre 1000 et 1700 ppm inclus (n1)
529  n2 = calculProportionHaute(); // nombre de valeurs supérieures à 1700 ppm (n2)
530  n0 = mesuresCO2.size() - n1 - n2;
531 
532  // Division par zéro ?
533  if((n0 + n1 + n2) == 0)
534  return;
535 
536  f1 = (double)n1 / (double)(n0 + n1 + n2);
537  f2 = (double)n2 / (double)(n0 + n1 + n2);
538 
539  qDebug() << Q_FUNC_INFO << "n0" << n0 << "n1" << n1 << "n2" << n2;
540  double calculIcone = (2.5 / log(2.)) * log(1. + f1 + 3. * f2);
541  qDebug() << Q_FUNC_INFO << "calculIcone" << calculIcone;
542 
543  if(calculIcone < SEUIL_ICONE_NUL)
544  {
545  indiceConfinement = IndiceConfinement::Nul;
546  }
547  else if((calculIcone >= SEUIL_ICONE_NUL) && calculIcone < SEUIL_ICONE_FAIBLE)
548  {
549  indiceConfinement = IndiceConfinement::Faible;
550  }
551  else if((calculIcone >= SEUIL_ICONE_FAIBLE) && calculIcone < SEUIL_ICONE_MOYEN)
552  {
553  indiceConfinement = IndiceConfinement::Moyen;
554  }
555  else if((calculIcone >= SEUIL_ICONE_MOYEN) && calculIcone < SEUIL_ICONE_ELEVE)
556  {
557  indiceConfinement = IndiceConfinement::Eleve;
558  }
559  else if((calculIcone >= SEUIL_ICONE_ELEVE) && calculIcone < SEUIL_ICONE_TRES_ELEVE)
560  {
561  indiceConfinement = IndiceConfinement::TresEleve;
562  }
563  else if(calculIcone >= SEUIL_ICONE_TRES_ELEVE)
564  {
565  indiceConfinement = IndiceConfinement::Extreme;
566  }
567  }
568  else
569  {
570  MesureCO2 mesureCO2 = getMesureCO2();
571  qDebug() << Q_FUNC_INFO << "mesureCO2" << mesureCO2.co2;
572 
573  if(mesureCO2.co2 >= SEUIL_MAX_CO2_CLASSE)
574  {
575  indiceConfinement = IndiceConfinement::Eleve;
576  }
577  else
578  {
579  indiceConfinement = IndiceConfinement::Nul;
580  }
581  }
582  qDebug() << Q_FUNC_INFO << "indiceConfinement" << indiceConfinement
583  << SalleEco::getIndiceConfinement(indiceConfinement);
584 
585  if(indiceConfinement != indiceConfinementPrecedent)
586  {
587  emit nouvelIndiceConfinement(nom, SalleEco::getIndiceConfinement(indiceConfinement));
588  }
589 }

591 void SalleEco::determinerIndiceIADI()
592 {
593  int indiceIADIPrecedent = indiceIADI;
594  // L’indice de confort thermique IADI (Indoor Air Disconfort Index) se calcule (Moschandreas
595  // et Sofuoglu, 2004) à partir de la température de l’air intérieur et l’humidité relative
596  qDebug() << Q_FUNC_INFO << "temperature" << getTemperature().temperature << "humidite"
597  << getHumidite().humidite;
598  double iadi = getTemperature().temperature - 0.55 * (1 - 0.01 * getHumidite().humidite) *
599  (getTemperature().temperature - 14.5);
600  qDebug() << Q_FUNC_INFO << "iadi" << iadi;
601 
602  if(iadi < SEUIL_INCONFORT_IADI_AUCUN)
603  {
604  indiceIADI = IndiceInconfortIADI::Aucun;
605  }
606  else if(iadi >= SEUIL_INCONFORT_IADI_AUCUN && iadi < SEUIL_INCONFORT_IADI_GENE)
607  {
608  indiceIADI = IndiceInconfortIADI::Gene;
609  }
610  else if(iadi >= SEUIL_INCONFORT_IADI_GENE && iadi < SEUIL_INCONFORT_IADI_MAL_ETRE)
611  {
612  indiceIADI = IndiceInconfortIADI::MalEtre;
613  }
614  else if(iadi >= SEUIL_INCONFORT_IADI_MAL_ETRE && iadi < SEUIL_INCONFORT_IADI_INCONFORT)
615  {
616  indiceIADI = IndiceInconfortIADI::Inconfort;
617  }
618  else if(iadi >= SEUIL_INCONFORT_IADI_INCONFORT && iadi < SEUIL_INCONFORT_IADI_STRESS_INTENSE)
619  {
620  indiceIADI = IndiceInconfortIADI::StressIntense;
621  }
622  else if(iadi >= SEUIL_INCONFORT_IADI_STRESS_INTENSE)
623  {
624  indiceIADI = IndiceInconfortIADI::UrgenceMedicale;
625  }
626 
627  qDebug() << Q_FUNC_INFO << "indiceIADI" << indiceIADI << SalleEco::getIndiceIADI(indiceIADI);
628 
629  if(indiceIADI != indiceIADIPrecedent)
630  {
631  emit nouvelIndiceIADI(nom, SalleEco::getIndiceIADI(indiceIADI));
632  }
633 }
